Code example

It's solving of the Collatz hypothesis

 static void Execute(int max_element)
 {
     int counter = 1;
     int max_counter = 1;
     int element = 1;
     int number;
     string result = "";
     string buffer = "";

     while (element < max_element)
     {
         number = element;
         do
         {
             buffer += number;
             buffer += " ";
             if (number % 2 == 0)
             {
                 number /= 2;
             }
             else
             {
                 number = number * 3 + 1;
             }
             counter++;
         } while (number != 1);
         if (counter > max_counter)
         {
             max_counter = counter;
             result = buffer;
         }
         buffer = "";
         counter = 1;
         element++;
      }

     result += "1";
     Console.WriteLine(result);
     Console.WriteLine("\nВ последовательности " + max_counter + 
     " символов.");
     Console.WriteLine();
 }
